```python
def solve():
    relations = [
        ("Alice works_at CompanyX", "CompanyX located_in Berlin"),
        ("Alice works_at CompanyY", "CompanyY located_in London")
    ]

    def find_all_paths(start, end, path=[]):
        path = path + [start]
        if start == end:
            return [path]
        paths = []
        for relation in relations:
            if relation[0] == start:
                new_path = find_all_paths(relation[1], relation[1], path)
                for p in new_path:
                    paths.append(p)
        return paths

    for path in find_all_paths("Alice", "CompanyX"):
        print(path)

solve()
```
